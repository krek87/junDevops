#навигация по директориям
pwd - показать мое текущее место положение
cd / - переход в корневую директорию
cd ~ - переход в домашнюю директорию пользователя
cd .. - переход надиректорию вверх
cd New\ dir - переход в директорию, название которой содержит несколько слов
# Если нажать tab после ввода начала названия диектории или файла, то программа дополнит его самостоятельно
# Если существует несколько файлов (директорий) с таким названием, то программа отобразит их все

#создание директоий
mkdir Mydir
mkdir dir1 dir 2 dir3 - создание нескольких директорий
mkdir "New dir" - если нужно название директории из нескольких слов
mkdir -p dir1/dir2/dir3 -создание вложенных друг в друга директорий (-P - parents или родительский)

#удаление директорий
rmdir Mydir
rmdir dir1 dir2 dir3
rmdir - удаляет только пустые директории
rm -r dir1 - удаляет директорию и все, что в нее вложено
