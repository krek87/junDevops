# 6. Стандартные потоки
# 1. Запишите в файл mykernel последние 10 строк файла /var/log/kern.log
sudo tail /var/log/kern.log > mykernel

# 2. Добавьте (не перезаписывайте) в этот файл с 13 по 21 строку из файла /etc/group
head -21 /etc/group | tail +13 >> mykernel

# 3. Посмотрите файлы mykernel и /etc/apg.conf, при этом стандартный вывод направьте в файл result.log, 
# а ошибки в файл /tmp/ERRORS
cat mykernel /etc/apg.cong > result.log 2> /tmp/ERRORS

# 4. Направьте строки с 13 до 20 из файла /etc/group в файл ~/lines, а потом добавьте в этот файл 16 
# строку из файла /etc/passwd.
head -20 /etc/group | tail +13 > ~/lines | head -16 /etc/passwd | tail +16 >> ~/lines
